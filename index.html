Stress Test — VaR Monte Carlo (Python)


import numpy as np
np.random.seed(42)

# Simulated daily returns (mean, std)
mu, sigma = 0.0005, 0.012
n = 252

# Portfolio P&L simulation
sim = np.random.normal(mu, sigma, (10000, n)).sum(axis=1)
pl = 1_000_000 * sim

# 99% Value at Risk (VaR)
var_99 = -np.percentile(pl, 1)
print(f"99% VaR: ${var_99:,.0f}")
